enum class EstadosBebes {
	despierto,
	llorando,
	dormido,
	comiendo,
	cagado_y_miado,
	viendo_paw_patrol,
	tieso,
	//comportamientos agregados
	tirando_putazos,
	reprobando_el_semestre,
	prime,
	vibrando_alto,
	sellando_la_hoonmoon,
	adominacion_total,
	licenciado_ingeniero
}

class Bebe(
	    var nombre: String,
	    var hambre: Int,
	    var pañal_lleno: Int,
	    var sueño: Int,
	    var enojo: Int,
	    var calificacion: Int,
	    var musica: String,
	    var tamaño_del_brazo :Int,
	    var estado: EstadosBebes = EstadosBebes.despierto,
	) {
//comportamientos agregados
	fun estetica() {
		if(estado== EstadosBebes.tieso)
		{
			estado == EstadosBebes.tieso
		}
		else if(tamaño_del_brazo >= 45)
		{
			estado = EstadosBebes.prime
		}
	}

	fun pelear()
	{
		if (estado == EstadosBebes.tieso || tamaño_del_brazo <=35)
		{
			estado = EstadosBebes.tieso
		}
		else if(enojo > 20)
		{
			enojo = 0;
			estado = EstadosBebes.tirando_putazos;
		}
		/*else if(enojo>20 && tamaño_del_brazo <=35)
		{
		    enojo = 0;
		    estado = EstadosBebes.tieso;
		}*/
	}

	fun escuchando_musica()
	{
		if((estado == EstadosBebes.despierto || estado == EstadosBebes.dormido || estado == EstadosBebes.tieso || estado == EstadosBebes.reprobando_el_semestre) && musica == "Angeles de la humanidad")
		{
			estado = EstadosBebes.vibrando_alto
		}
		else if((estado == EstadosBebes.despierto || estado == EstadosBebes.dormido || estado == EstadosBebes.tieso) && musica == "Golden")
		{
			estado = EstadosBebes.despierto
			         estado = EstadosBebes.sellando_la_hoonmoon
		}
		else if(nombre == "Ado" && musica == "Usseewa")
		{
		    estado = EstadosBebes.despierto
            estado = EstadosBebes.adominacion_total
		}
	}

	fun estudiar()
	{
		if(estado == EstadosBebes.tieso)
		{
			estado = EstadosBebes.tieso
		}
		else if(calificacion <= 69)
		{
			estado = EstadosBebes.reprobando_el_semestre
		}
		else if(calificacion >= 70)
		{
			estado = EstadosBebes.licenciado_ingeniero
		}
	}

	fun cambio()
	{
		if(estado == EstadosBebes.tieso)
		{
			estado = EstadosBebes.tieso
		}
		else if(pañal_lleno <= 30)
		{

		}
		else if(pañal_lleno >= 70)
		{
			estado = EstadosBebes.cagado_y_miado
		}
	}
/*-----------------------------------------------------------------------------*/
	fun alimentar() {
		if(estado == EstadosBebes.tieso)
		{
			estado = EstadosBebes.tieso
		}
		else
		{
			estado = EstadosBebes.comiendo
		}
	}

	fun dormir() {
		if((estado == EstadosBebes.despierto || estado == EstadosBebes.comiendo) && sueño > 49) {
			sueño = 0
			         estado = EstadosBebes.dormido
		}
	}

	fun despertar() {
		if(estado == EstadosBebes.dormido) {
			estado = EstadosBebes.despierto
		}
	}

	fun despertar_con_agua() {
		if(estado == EstadosBebes.dormido) {
			estado = EstadosBebes.tieso
		}
	}

	fun actualizar_estado() {
		when(estado) {
			EstadosBebes.despierto -> {
				sueño += 5
				hambre += 5
				pañal_lleno += 3
			}

			EstadosBebes.comiendo -> {
				hambre -= 40
			}

			else -> {

				}
			}
	}

	fun picar_con_un_palo(): String{
		return estado_actual()
	}

	fun estado_actual(): String{
		when(estado) {
		//comportamientos agregados
		/*-----------------------------------------------------------------------------*/
			EstadosBebes.prime -> {
				return "Ocho palabras furro de estetica miserable, 'ESTETICA', de la mas alta calidad"
			}
			EstadosBebes.vibrando_alto -> {
				return "EN LOS FUEGOS DE LA BATALLA, EN EL YUNQUE DE LA GUERRA, VULKAN VIVE!!!"
			}
			EstadosBebes.adominacion_total -> {""
				return "HAAA? USSEE USSEE USSEE WA, ANATA GA OMOU YORI KENKOU  DESU, ISSAI GASSAI BONYOU NA, ANATA JA WAKARANAI KAMO NE!!"
			}
			EstadosBebes.sellando_la_hoonmoon -> {
				return "WE ARE GOING UP, UP, UP ITS OUR MOMENT, YOU KNOW  TOGETHER WE'RE GLOWIN' GONNA BE, GONNA BE GOLDEN"
			}
			EstadosBebes.tirando_putazos -> {
				return "QUE SE ARMEN LOS PINCHES CHINGADAZOS!!"
			}
			EstadosBebes.reprobando_el_semestre -> {
				return "El bebe mejor va a darse de baja y poner un puesto de pollos y carnitas"
			}
			EstadosBebes.licenciado_ingeniero -> {
				return "El bebe se ah graduado y licenciado con completo exito de diseC1o, ahora no encuentra chamba de lo que estudio...mejor hubiera abierto un puesto de carnitas"
			}
			EstadosBebes.cagado_y_miado -> {
				return "Al bebe se le esta desbordando el relleno"
			}
			EstadosBebes.llorando -> {
				return "El bebe esta llorando porque no lo topa ni su madre"
			}
			EstadosBebes.despierto -> {
				return "El bebe esta despierto"
			}
			EstadosBebes.tieso -> {
				return "Estoy muerto"
			}
			EstadosBebes.dormido -> {
				return "Esta dormidito"
			}
			EstadosBebes.comiendo -> {
				return "Estoy alimentadome para ser grande y fuerte"
			}
			else -> {
				    return "No identifico el estado"
				}
			}
	}
}
