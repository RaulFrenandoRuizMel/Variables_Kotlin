class Jeager
	(
	    val categoria: Int,
	    var nombre: String,
	    var piloto1: String,
	    var piloto2: String,
	    val pais_de_origen: String,
	    val Armamento: String,
	) {
	fun presentar_pilotos()
	{
		println("Los pilotos del Jeager: ${nombre}, categoria ${categoria} originario de ${pais_de_origen} son:")
		println("${piloto1} y ${piloto2}");
		println("Cuentan con Armamento de: ${Armamento}");
		println("");
	}
}

class Kaiju
	(
	    val categoria: Int,
	    var nombre: String,
	    val lugar: String,
	    val Habilidad: String,
	    val peso: Int,
	    val tamaño: Int
	) {
	fun informacion_Kaiju()
	{
		println("El kaiju de categoria ${categoria}, de nombre ${nombre} ha llegado a ${lugar}, cuenta con la habilidad de: ${Habilidad}")
		println("Peso: ${peso} toneladas")
		println("Tamaño: ${tamaño} Metros")
		println("");
	}
}

class Shaterdomme
	(
	    val capacidad: Int,
	    var ubicacion: String,
	    var director: String,
	) {
	fun descripcion()
	{
		println("El Shaterdomme de ${ubicacion} a cargo de ${director} cuenta con capacidad de ${capacidad} Jeagers")
		println("");
	}
}

class Piloto
	(
	    val edad: Int,
	    var nombre: String,
	    var nacionalidad: String,
	    var experiencia: String,
	) {
	fun datos_pilotos()

		println("nombre: ${nombre}")
        println("edad: ${edad} años")
        println("nacionalidad: ${nacionalidad}")
        println("experiencia: ${experiencia} años")
		println("");
	}
}

		}

    	}

}
